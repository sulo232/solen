let lang = 'de';
let currentUser = null;
let ownerStoreId = null;
let activeCat = 'all';
let activeDetailId = null;

function t(key) { return T[lang][key] || key; }

// ═══════════════════════════════════════════════════════
// LANG
// ═══════════════════════════════════════════════════════
function setLang(l) {
  lang = l;
  document.querySelectorAll('.lang-btn').forEach(b => b.classList.toggle('active', b.textContent === l.toUpperCase()));
  applyTranslations();
  renderGrid();
}

function detectLang() {
  const nav = navigator.language || navigator.userLanguage || 'de';
  lang = nav.startsWith('de') ? 'de' : 'en';
  document.querySelectorAll('.lang-btn').forEach(b => b.classList.toggle('active', b.textContent === lang.toUpperCase()));
}

function applyTranslations() {
  document.querySelectorAll('[data-t]').forEach(el => {
    const key = el.getAttribute('data-t');
    if (!T[lang][key]) return;
    if (el.innerHTML.includes('<')) {
      el.innerHTML = T[lang][key];
    } else {
      el.textContent = T[lang][key];
    }
  });
}

// ═══════════════════════════════════════════════════════
// GRID
// ═══════════════════════════════════════════════════════
function renderGrid() {
  const grid = document.getElementById('listingGrid');
  const items = activeCat === 'all' ? LISTINGS : LISTINGS.filter(l => l.cat === activeCat);
  grid.innerHTML = items.map(l => cardHTML(l)).join('');
}

function catLabel(cat) {
  return {hair: t('cat_hair'), barber: t('cat_barber'), nails: t('cat_nails')}[cat] || cat;
}

function cardHTML(l) {
  const isOwner = ownerStoreId === l.id;
  const ownerBtn = isOwner ? `<button class="card-owner-btn" onclick="event.stopPropagation();openEdit(${l.id})">${t('edit_btn')}</button>` : '';
  const badgeClass = l.registered ? 'badge-registered' : 'badge-unregistered';
  const badgeText = l.registered ? '✓ Bookable' : 'View Only';
  const actionHTML = l.registered
    ? `<button class="btn btn-primary btn-sm" onclick="event.stopPropagation();openBookingDirect(${l.id})">${t('book_btn')}</button>`
    : `<div class="card-not-bookable">${t('not_bookable')}</div>`;
  return `
  <div class="card" onclick="openDetail(${l.id})">
    <div class="card-img">
      <img src="${l.photos[0]}" alt="${l.name}" loading="lazy">
      <span class="card-badge ${badgeClass}">${badgeText}</span>
      ${ownerBtn}
    </div>
    <div class="card-body">
      <div class="card-cat">${catLabel(l.cat)}</div>
      <div class="card-name">${l.name}</div>
      <div class="card-location">📍 ${l.address}</div>
      <div class="card-meta">
        <div class="card-rating"><span class="stars">★</span>${l.rating} <span style="color:var(--mid);font-weight:400">(${l.reviews})</span></div>
        <div class="card-price">${l.price}</div>
      </div>
      <div class="card-action">${actionHTML}</div>
    </div>
  </div>`;
}

function filterCat(cat) {
  activeCat = cat;
  document.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active'));
  const map = {all:'filter_all',hair:'filter_hair',barber:'filter_barber',nails:'filter_nails'};
  document.querySelector(`[data-t="${map[cat]}"]`)?.classList.add('active');
  renderGrid();
  if (cat !== 'all') document.getElementById('listings').scrollIntoView({behavior:'smooth'});
}

function openDetail(id) {
  const l = LISTINGS.find(x => x.id === id);
  if (!l) return;
  activeDetailId = id;
  galleryIndex = 0;

  // Gallery
  const gallery = document.getElementById('detailGallery');
  // Remove old images
  gallery.querySelectorAll('.gallery-img').forEach(i => i.remove());
  l.photos.forEach((src, i) => {
    const img = document.createElement('img');
    img.className = 'gallery-img' + (i === 0 ? ' active' : '');
    img.src = src;
    img.alt = l.name;
    gallery.insertBefore(img, gallery.querySelector('.gallery-arrow'));
  });
  renderGalleryDots(l);

  // Header
  document.getElementById('detailCat').textContent = catLabel(l.cat);
  document.getElementById('detailName').textContent = l.name;
  document.getElementById('detailRating').textContent = `${l.rating} (${l.reviews} Reviews)`;

  // Action buttons
  const actionsEl = document.getElementById('detailModalActions');
  if (l.registered) {
    actionsEl.innerHTML = `<button class="btn btn-primary" onclick="openBookingDirect(${l.id})">${t('book_btn')}</button>`;
  } else {
    actionsEl.innerHTML = `<div style="font-size:12px;color:var(--mid);max-width:180px;text-align:right">${t('not_bookable')}</div>`;
  }
  if (ownerStoreId === l.id) {
    actionsEl.innerHTML += `<button class="btn btn-outline btn-sm" onclick="openEdit(${l.id})">${t('edit_btn')}</button>`;
  }

  // Info chips
  const chips = document.getElementById('detailChips');
  chips.innerHTML = '';
  if (l.phone) chips.innerHTML += `<a class="chip" href="tel:${l.phone}"><span class="chip-icon">📞</span>${l.phone}</a>`;
  if (l.address) chips.innerHTML += `<a class="chip" href="https://maps.google.com?q=${encodeURIComponent(l.address)}" target="_blank"><span class="chip-icon">📍</span>${l.address}</a>`;
  if (l.website) chips.innerHTML += `<a class="chip" href="${l.website}" target="_blank"><span class="chip-icon">🌐</span>${lang==='de'?'Webseite':'Website'}</a>`;
  if (l.insta) chips.innerHTML += `<a class="chip" href="https://instagram.com/${l.insta.replace('@','')}" target="_blank"><span class="chip-icon">📸</span>${l.insta}</a>`;

  // Services tab
  document.getElementById('detailServices').innerHTML = l.services.map(s => `
    <div class="service-row">
      <div class="service-name">${lang === 'de' ? s.name : (s.nameEN || s.name)}</div>
      <div class="service-details">
        <span class="service-dur">⏱ ${s.duration}</span>
        <span class="service-price">${s.price}</span>
      </div>
    </div>`).join('');

  // Hours tab
  const daysDE = {Mon:'Mo',Tue:'Di',Wed:'Mi',Thu:'Do',Fri:'Fr',Sat:'Sa',Sun:'So'};
  const daysEN = {Mon:'Mon',Tue:'Tue',Wed:'Wed',Thu:'Thu',Fri:'Fri',Sat:'Sat',Sun:'Sun'};
  const days = lang === 'de' ? daysDE : daysEN;
  document.getElementById('detailHours').innerHTML = Object.entries(l.hours).map(([d,h]) => {
    const isClosed = h === 'Closed';
    return `<div class="hours-row">
      <div class="hours-day">${days[d]}</div>
      <div class="hours-time ${isClosed ? 'hours-closed' : ''}">${isClosed ? t('closed') : h}</div>
    </div>`;
  }).join('');

  // Info tab
  document.getElementById('detailDesc').textContent = lang === 'de' ? l.desc : (l.descEN || l.desc);
  document.getElementById('detailAddress').textContent = l.address;

  // Reset to first tab
  document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
  document.querySelectorAll('.tab-panel').forEach(p => p.classList.remove('active'));
  document.querySelector('.tab').classList.add('active');
  document.getElementById('tab-services').classList.add('active');

  openModal('detailOverlay');
}

function renderGalleryDots(l) {
  const dots = document.getElementById('galleryDots');
  dots.innerHTML = l.photos.map((_, i) =>
    `<button class="gallery-dot ${i===0?'active':''}" onclick="goGallery(${i})"></button>`
  ).join('');
  document.getElementById('galleryCount').textContent = `1 / ${l.photos.length}`;
}

function galleryNav(dir) {
  const l = LISTINGS.find(x => x.id === activeDetailId);
  if (!l) return;
  galleryIndex = (galleryIndex + dir + l.photos.length) % l.photos.length;
  goGallery(galleryIndex);
}

function goGallery(i) {
  const l = LISTINGS.find(x => x.id === activeDetailId);
  if (!l) return;
  galleryIndex = i;
  document.querySelectorAll('#detailGallery .gallery-img').forEach((img, idx) => img.classList.toggle('active', idx === i));
  document.querySelectorAll('#galleryDots .gallery-dot').forEach((dot, idx) => dot.classList.toggle('active', idx === i));
  document.getElementById('galleryCount').textContent = `${i + 1} / ${l.photos.length}`;
}

function switchTab(name) {
  document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
  document.querySelectorAll('.tab-panel').forEach(p => p.classList.remove('active'));
  document.querySelector(`[onclick="switchTab('${name}')"]`).classList.add('active');
  document.getElementById(`tab-${name}`).classList.add('active');
}

function openBookingDirect(id) {
  const l = LISTINGS.find(x => x.id === id);
  if (!l) return;
  document.getElementById('bookingCat').textContent = catLabel(l.cat);
  document.getElementById('bookingName').textContent = l.name;
  document.getElementById('bookingSub').textContent = t('book_sub');

  const sel = document.getElementById('bService');
  sel.innerHTML = l.services.map(s => `<option value="${lang==='de'?s.name:s.nameEN||s.name}">${lang==='de'?s.name:s.nameEN||s.name} — ${s.price}</option>`).join('');

  // Set min date to today
  const today = new Date().toISOString().split('T')[0];
  document.getElementById('bDate').min = today;
  document.getElementById('bDate').value = today;

  // Pre-fill from current user
  if (currentUser) {
    document.getElementById('bName').value = currentUser.name || '';
    document.getElementById('bEmail').value = currentUser.email || '';
  }

  // Store current salon for booking
  document.getElementById('bookingForm').dataset.salonId = id;

  if (document.getElementById('detailOverlay').classList.contains('open')) {
    closeModal('detailOverlay');
    setTimeout(() => openModal('bookingOverlay'), 200);
  } else {
    openModal('bookingOverlay');
  }
}

async function confirmBooking() {
  const salonId = parseInt(document.getElementById('bookingForm').dataset.salonId);
  const salon = LISTINGS.find(x => x.id === salonId);
  const customerName = document.getElementById('bName').value.trim();
  const customerEmail = document.getElementById('bEmail').value.trim();
  const service = document.getElementById('bService').value;
  const date = document.getElementById('bDate').value;
  const time = document.getElementById('bTime').value;

  if (!customerName || !customerEmail || !date) {
    showToast('⚠️ ' + (lang === 'de' ? 'Bitte alle Felder ausfüllen.' : 'Please fill in all fields.'));
    return;
  }

  const btn = document.querySelector('#bookingForm .btn-primary');
  btn.textContent = '...';
  btn.disabled = true;

  try {
    await fetch('/api/send-email', {
      method: 'POST',
      headers: {'Content-Type': 'application/json'},
      body: JSON.stringify({
        customerEmail, customerName,
        salonName: salon.name,
        salonEmail: salon.ownerEmail || '',
        serviceName: service,
        date, time, lang
      })
    });
  } catch(e) {
    console.log('Email not available in preview mode');
  }

  showToast(t('booking_success'));
  closeModal('bookingOverlay');
  btn.textContent = t('form_book');
  btn.disabled = false;
}

function openEdit(id) {
  const l = LISTINGS.find(x => x.id === id);
  if (!l || ownerStoreId !== id) return;

  document.getElementById('editName').value = l.name;
  document.getElementById('editPhone').value = l.phone || '';
  document.getElementById('editWebsite').value = l.website || '';
  document.getElementById('editInsta').value = l.insta || '';
  document.getElementById('editDesc').value = l.desc || '';
  document.getElementById('editDescEN').value = l.descEN || '';
  renderEditPhotos(l);

  document.getElementById('editOverlay').dataset.editId = id;
  if (document.getElementById('detailOverlay').classList.contains('open')) {
    closeModal('detailOverlay');
    setTimeout(() => openModal('editOverlay'), 200);
  } else {
    openModal('editOverlay');
  }
}

function renderEditPhotos(l) {
  const grid = document.getElementById('editPhotoGrid');
  grid.innerHTML = l.photos.map((src, i) => `
    <div class="photo-thumb">
      <img src="${src}" alt="">
      <button class="photo-remove" onclick="removeEditPhoto(${i})">✕</button>
    </div>`).join('');
  grid.innerHTML += `<div class="photo-add" onclick="document.getElementById('editPhotoUrl').focus()">＋<span>Add photo</span></div>`;
}

function removeEditPhoto(idx) {
  const id = parseInt(document.getElementById('editOverlay').dataset.editId);
  const l = LISTINGS.find(x => x.id === id);
  if (!l || l.photos.length <= 1) return;
  l.photos.splice(idx, 1);
  renderEditPhotos(l);
}

function addEditPhoto() {
  const url = document.getElementById('editPhotoUrl').value.trim();
  if (!url) return;
  const id = parseInt(document.getElementById('editOverlay').dataset.editId);
  const l = LISTINGS.find(x => x.id === id);
  if (!l) return;
  l.photos.push(url);
  renderEditPhotos(l);
  document.getElementById('editPhotoUrl').value = '';
}

function saveEdit() {
  const id = parseInt(document.getElementById('editOverlay').dataset.editId);
  const l = LISTINGS.find(x => x.id === id);
  if (!l) return;
  l.name = document.getElementById('editName').value.trim() || l.name;
  l.phone = document.getElementById('editPhone').value.trim();
  l.website = document.getElementById('editWebsite').value.trim();
  l.insta = document.getElementById('editInsta').value.trim();
  l.desc = document.getElementById('editDesc').value.trim();
  l.descEN = document.getElementById('editDescEN').value.trim();
  closeModal('editOverlay');
  renderGrid();
  showToast(t('save_success'));
}
